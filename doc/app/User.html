def self.create(device_type)
  user = User.new
  user.device_type = device_type
  user.session = User.create_session()

  if user.save() then
    return user
  else
    return nil
  end
end

def self.is_invalid(user_id, api_session)
  user = User.find_by_id_and_session(user_id, api_session)
  return user.nil?
end
